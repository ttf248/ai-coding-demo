class Person {
  name: string
  age: number

  constructor(name: string, age: number) {
    this.name = name
    this.age = age
  }

  greet(): string {
    return `Hello, I'm ${this.name}`
  }
}

class Employee extends Person {
  employeeId: number
  department: string

  constructor(
    name: string,
    age: number,
    id: number,
    dept: string
  ) {
    super(name, age)
    this.employeeId = id
    this.department = dept
  }

  greet(): string {
    return `${super.greet()}, ID: ${this.employeeId}`
  }

  getInfo(): string {
    return `${this.name}, ${this.age}, Dept: ${this.department}`
  }
}
